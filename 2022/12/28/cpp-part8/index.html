// template 动态增长万能模板
void test01() {
    vector v;
    for (int i = 0; i < 10; i++)
        v.push_back(i);
    cout << v[3] << endl;
    // iterator 迭代器模式
    vector::iterator it = v.begin();
    while (it != v.end()) {
        cout << *it << endl;
        it++;
    }
    // iterator 对各容器通用
}
